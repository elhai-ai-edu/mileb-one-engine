const { GoogleGenerativeAI } = require("@google/generative-ai");

exports.handler = async (event) => {

  const { message, studentStep } = JSON.parse(event.body);

  const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY);

  const model = genAI.getGenerativeModel({ model: "gemini-1.5-flash" });

  // הגדרת הנחיות לפי השלב ב-Firebase

  let systemInstruction = "";

  if (studentStep === 1) {

    systemInstruction = "עזור לסטודנט לאתר בעיה למו\"פ. שאל על הלקוח ופתרונות עבר לפי המסמך.";

  } else if (studentStep === 2) {

    systemInstruction = "הסטודנט עבר שלב! כעת עזור לו למצוא 2 פתרונות באתרים מהימנים כמו מכון דוידסון.";

  }

  const chat = model.startChat({

    history: [

      { role: "user", parts: [{ text: systemInstruction }] },

      { role: "model", parts: [{ text: "שלום, אני המלווה שלך לפרויקט. בוא נתחיל בשלב הנוכחי." }] }

    ],

  });

  const result = await chat.sendMessage(message);

  return {

    statusCode: 200,

    body: JSON.stringify({ response: result.response.text() }),

  };

};

## **‫Gemini אמר**

הקובץ ששיתפת מבהיר שהמערכת שלך בנויה בצורה היררכית חכמה (Architecture 4.2), שבה ה**Config** (ה-JSON) הוא ה"מוח", והקוד הוא רק ה"מנוע" שמריץ אותו.

זה חדשות מצוינות, כי זה אומר שכדי להטמיע את "פרויקט הגמר בעברית", אתה לא צריך לשנות את התשתית הטכנית, אלא רק להוסיף אובייקטים חדשים ל-Config הקיים.

הנה איך זה משתלב במבנה של **MilEd.One**:

### **1. עדכון ה-Config (הגדרת הבוטים)**

בתוך רשימת ה-branches.courses.items, כבר מופיע הבוט hebrew_optometry_final. כעת, עלינו להחליף את ה-systemPrompt הגנרי ב"חוקה" וב"משימות" שבנינו, תוך שימוש במנגנון השכבות (_layer) שלך.

במקום בוט אחד, כדאי להגדיר ב-JSON **שלוש משימות (Tasks)** או לעדכן את הפרומפט שיישאב את הנתונים לפי ה-Phase:

### **2. שילוב ה-Kernel הפדגוגי**

ה-Config שלך כולל kernel עם חוקים נוקשים ל-development. זה מושלם עבורנו:

- **noFullSolution: true:** מבטיח שהבוט לא יכתוב לסטודנט את פסקת 200 המילים בעצמו, אלא רק ינחה אותו.
- **effortRegulation: strict:** מחייב את הסטודנט להתאמץ לפני שהוא מקבל אישור מעבר.

### **3. היישום הטכנולוגי (Firebase + Netlify)**

בגלל שהמערכת שלך כבר עובדת עם מנוע מרכזי, הפעולות הנדרשות הן:

- **ב-GitHub:** הוספת קובץ JSON חדש (או עדכון הקיים) שבו ה-System Prompt של הבוט hebrew_optometry_final מכיל את הדירוג (1-10) והמחוון שבנינו.
- **ב-Firebase:** המערכת שלך בטח כבר שומרת botType. עליך להוסיף שדה userProgress שמתעד את הציונים (1-10) שה-AI החזיר.
- **ב-Netlify:** הפונקציה הקיימת שלך (שמייצאת את הפרומפט ב-merged mode) תדע לקחת את ה-Kernel, להוסיף לו את ה"חוקה" ואת ה"משימה" הספציפית לסטודנט.

### **מה זה אומר על המורכבות?**

עבורך, המורכבות יורדת כמעט לאפס. **אתה לא בונה מערכת חדשה**, אתה פשוט מוסיף "תוכן" למערכת קיימת.

האם תרצה שאכתוב לך את הבלוק המלא של ה-JSON (תחת hebrew_optometry_final) בפורמט המדויק של המערכת שלך (גרסה 4.2), כך שתוכל פשוט "להדביק" אותו ב-Config שלך?

// netlify/functions/submit.js — MilEd.One

// Stores student submissions (assignments / artifacts)

import { initializeApp, getApps, cert } from "firebase-admin/app";

import { getDatabase } from "firebase-admin/database";

// ─────────────────────────────────────────

// FIREBASE INIT

// ─────────────────────────────────────────

function getDB(){

  if(!getApps().length){

    const serviceAccount =

      JSON.parse(process.env.FIREBASE_SERVICE_ACCOUNT);

    initializeApp({

      credential: cert(serviceAccount),

      databaseURL: process.env.FIREBASE_DB_URL

    });

  }

  return getDatabase();

}

// ─────────────────────────────────────────

// HEADERS

// ─────────────────────────────────────────

const headers = {

  "Access-Control-Allow-Origin":"*",

  "Access-Control-Allow-Headers":"Content-Type",

  "Access-Control-Allow-Methods":"POST, OPTIONS",

  "Content-Type":"application/json"

};

// ─────────────────────────────────────────

// HANDLER

// ─────────────────────────────────────────

export async function handler(event){

  if(event.httpMethod === "OPTIONS"){

    return { statusCode:200, headers, body:"" };

  }

  if(event.httpMethod !== "POST"){

    return {

      statusCode:405,

      headers,

      body:JSON.stringify({error:"Method not allowed"})

    };

  }

  try{

    const body = JSON.parse(event.body || "{}");

    const submission = body.submission || {};

    const studentId   = submission.studentId;

    const classId     = submission.classId;

    const sessionId   = submission.sessionId || null;

    const botType     = submission.botType   || null;

    const assignmentId = submission.assignmentId || "general";

    const step         = submission.step || 1;

    const version      = submission.version || 1;

    const content     = submission.content;

    if(!studentId || !classId || !content){

      return {

        statusCode:400,

        headers,

        body:JSON.stringify({

          error:"studentId, classId and content are required"

        })

      };

    }

    const db = getDB();

    const submissionId =

      "sub_" + Date.now() + "_" + Math.random().toString(36).slice(2,6);

    const timestamp = Date.now();

    const record = {

      submissionId,

      studentId,

      classId,

      sessionId,

      botType,

      assignmentId,

      step,

      version,

      content,

      timestamp

    };

    await db

      .ref(`submissions/${classId}/${studentId}/${submissionId}`)

      .set(record);

    console.log("SUBMISSION SAVED:",record);

    return {

      statusCode:200,

      headers,

      body:JSON.stringify({

        ok:true,

        submissionId

      })

    };

  }

  catch(err){

    console.error("SUBMIT ERROR:",err);

    return {

      statusCode:500,

      headers,

      body:JSON.stringify({

        error:"submission failed"

      })

    };

  }

}

## **‫Gemini אמר**

הקובץ submit.js ששיתפת הוא ה"חותמת" הסופית של התהליך. הוא משלים את התמונה הטכנולוגית בכך שהוא מנהל את הרישום הרשמי ב-Firebase Realtime Database בכל פעם שסטודנט לוחץ על כפתור השליחה.

במערכת **MilEd.One**, הקובץ הזה משמש כצומת שבו התוצר הופך מ"טיוטה" (Draft) ל"הגשה" (Submission). הנה איך הוא משתלב בפרויקט הגמר בעברית:

### **1. תיעוד הציונים (הערך המוסף)**

כדי שהמרצה (אתה) יוכל לראות לא רק את הטקסט, אלא גם את **הציון** שה-AI נתן, כדאי לעדכן את ה-Object שנשלח מה-Workspace לקובץ ה-submit.js.

בקובץ ששלחת, ה-record שומר כרגע content ו-step. אם נוסיף לו את ה-evaluation (ה-JSON שחזר מה-AI), יהיה לך דשבורד מלא של התקדמות הסטודנטים ב-Firebase.

### **2. הקשר ל-**classId** ו-**assignmentId

הקוד שלך מחלק את ההגשות בתיקיות לפי:

submissions / {classId} / {studentId} / {submissionId}

זה מצוין, כי זה מאפשר לך:

- **למרצים:** לראות את כל הסטודנטים של קורס מסוים במקום אחד.
- **לסטודנטים:** לשלוח מספר גרסאות (Versions) של אותו שלב, כשהמערכת מתעדת את כולן עם Timestamp.

### **סיכום המבנה המלא של המערכת שלך (End-to-End)**

הנה "מפת הדרכים" הסופית של המערכת לאחר כל הקבצים שניתחנו:

- **ממשק (**workspace.html**):** הסטודנט כותב בתיבת הטקסט ומנהל שיחה עם ה-AI.
- **מנוע (**chat.js**):** ה-AI משתמש ב-kernel.txt וב-config.json כדי להנחות את הסטודנט סוקרטית.
- **הערכה (**evaluate** / **chat**):** ה-AI בודק את הטקסט לפי המחוון (1-10) ומחזיר JSON עם סטטוס "מאושר/נדחה".
- **רישום (**submit.js**):** ברגע שהשלב מאושר, הנתונים נרשמים ב-Firebase תחת submissions, כולל הטקסט הסופי והציון.
- **מעבר שלב:** האתר מעדכן את ה-UI לשלב הבא (מ-1 עד 4).

### **המלצה אחרונה לביצוע (Implementation Tip):**

מכיוון שהקובץ submit.js כבר עובד, אתה רק צריך לוודא שבקובץ workspace.html, הפונקציה submitWork() קוראת לכתובת ה-URL הנכונה (/.netlify/functions/submit).

כך נראה האובייקט שאתה שולח לו:

**המערכת שלך עכשיו שלמה, מקצועית ותואמת את הסטנדרטים הגבוהים של MilEd.One v5.0.**

function renderDashboard(data) {

  const grid = document.getElementById('studentGrid');

  const students = data.students || [];

  document.getElementById('studentCount').textContent = students.length;

  if (students.length === 0) {

    grid.innerHTML = `<div class="empty-state"><div style="font-size:2rem">⏳</div>

      <p>ממתין לתלמידים להתחבר...</p></div>`;

    return;

  }

  const locked = data.lockedSteps || [];

  grid.innerHTML = students.map(s => {

    const isLocked  = locked.includes(s.studentId);

    const hasAnswer = s.steps && s.steps.length > 0;

    const cardClass = isLocked ? "locked" : hasAnswer ? "has-answer" : "";

    const statusClass = isLocked ? "status-locked" : hasAnswer ? "status-done" : "status-waiting";

    const statusText  = isLocked ? "🔒 נעול" : hasAnswer ? "✅ הגיש" : "⏳ ממתין";

    const stepsHtml = (s.steps || []).map(st => `

      <div class="step-item">

        <div class="step-label">שלב ${st.step}</div>

        <div class="step-content">${escapeHtml(st.content || "")}</div>

      </div>

    `).join('');

    return `

      <div class="student-card ${cardClass}">

        <div class="student-top">

          <span class="student-id">👤 ${s.studentId.slice(0,10)}</span>

          <span class="student-status ${statusClass}">${statusText}</span>

        </div>

        <div class="steps-list">${stepsHtml || '<div style="color:#aaa;font-size:.8rem">טרם הגיש</div>'}</div>

        <div class="card-actions">

          <button class="card-btn card-btn-unlock" onclick="unlockStudent('${s.studentId}')">🔓 פתח שלב</button>

          <button class="card-btn card-btn-lock"   onclick="lockStudent('${s.studentId}')">🔒 נעל</button>

        </div>

      </div>`;

  }).join('');

}
